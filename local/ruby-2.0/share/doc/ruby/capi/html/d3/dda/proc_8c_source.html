129 /*
130  * call-seq:
131  * prc.lambda? -> true or false
132  *
133  * Returns +true+ for a Proc object for which argument handling is rigid.
134  * Such procs are typically generated by +lambda+.
135  *
136  * A Proc object generated by +proc+ ignores extra arguments.
137  *
138  * proc {|a,b| [a,b] }.call(1,2,3) #=> [1,2]
139  *
140  * It provides +nil+ for missing arguments.
141  *
142  * proc {|a,b| [a,b] }.call(1) #=> [1,nil]
143  *
144  * It expands a single array argument.
145  *
146  * proc {|a,b| [a,b] }.call([1,2]) #=> [1,2]
147  *
148  * A Proc object generated by +lambda+ doesn't have such tricks.
149  *
150  * lambda {|a,b| [a,b] }.call(1,2,3) #=> ArgumentError
151  * lambda {|a,b| [a,b] }.call(1) #=> ArgumentError
152  * lambda {|a,b| [a,b] }.call([1,2]) #=> ArgumentError
153  *
154  * Proc#lambda? is a predicate for the tricks.
155  * It returns +true+ if no tricks apply.
156  *
157  * lambda {}.lambda? #=> true
158  * proc {}.lambda? #=> false
159  *
160  * Proc.new is the same as +proc+.
161  *
162  * Proc.new {}.lambda? #=> false
163  *
164  * +lambda+, +proc+ and Proc.new preserve the tricks of
165  * a Proc object given by <code>&</code> argument.
166  *
167  * lambda(&lambda {}).lambda? #=> true
168  * proc(&lambda {}).lambda? #=> true
169  * Proc.new(&lambda {}).lambda? #=> true
170  *
171  * lambda(&proc {}).lambda? #=> false
172  * proc(&proc {}).lambda? #=> false
173  * Proc.new(&proc {}).lambda? #=> false
174  *
175  * A Proc object generated by <code>&</code> argument has the tricks
176  *
177  * def n(&b) b.lambda? end
178  * n {} #=> false
179  *
180  * The <code>&</code> argument preserves the tricks if a Proc object
181  * is given by <code>&</code> argument.
182  *
183  * n(&lambda {}) #=> true
184  * n(&proc {}) #=> false
185  * n(&Proc.new {}) #=> false
186  *
187  * A Proc object converted from a method has no tricks.
188  *
189  * def m() end
190  * method(:m).to_proc.lambda? #=> true
191  *
192  * n(&method(:m)) #=> true
193  * n(&method(:m).to_proc) #=> true
194  *
195  * +define_method+ is treated the same as method definition.
196  * The defined method has no tricks.
197  *
198  * class C
199  * define_method(:d) {}
200  * end
201  * C.new.d(1,2) #=> ArgumentError
202  * C.new.method(:d).to_proc.lambda? #=> true
203  *
204  * +define_method+ always defines a method without the tricks,
205  * even if a non-lambda Proc object is given.
206  * This is the only exception for which the tricks are not preserved.
207  *
208  * class C
209  * define_method(:e, &proc {})
210  * end
211  * C.new.e(1,2) #=> ArgumentError
212  * C.new.method(:e).to_proc.lambda? #=> true
213  *
214  * This exception insures that methods never have tricks
215  * and makes it easy to have wrappers to define methods that behave as usual.
216  *
217  * class C
218  * def self.def2(name, &body)
219  * define_method(name, &body)
220  * end
221  *
222  * def2(:f) {}
223  * end
224  * C.new.f(1,2) #=> ArgumentError
225  *
226  * The wrapper <i>def2</i> defines a method which has no tricks.
227  *
228  */

935 static VALUE
936 mnew(VALUE klass, VALUE obj, ID id, VALUE mclass, int scope)
937 {

939  VALUE rclass = klass, defined_class;
940  ID rid = id;
941  struct METHOD *data;
942  rb_method_entry_t *me, meb;
943  rb_method_definition_t *def = 0;
944  rb_method_flag_t flag = NOEX_UNDEF;
945 
946  again:
947  me = rb_method_entry_without_refinements(klass, id, &defined_class);
948  if (UNDEFINED_METHOD_ENTRY_P(me)) {
949  ID rmiss = idRespond_to_missing;
950  VALUE sym = ID2SYM(id);
951 
952  if (obj != Qundef && !rb_method_basic_definition_p(klass, rmiss)) {
953  if (RTEST(rb_funcall(obj, rmiss, 2, sym, scope ? Qfalse : Qtrue))) {
954  def = ALLOC(rb_method_definition_t);
955  def->type = VM_METHOD_TYPE_MISSING;
956  def->original_id = id;
957  def->alias_count = 0;
958 
959  meb.flag = 0;
960  meb.mark = 0;
961  meb.called_id = id;
962  meb.klass = klass;
963  meb.def = def;
964  me = &meb;
965  def = 0;
966 
967  goto gen_method;
968  }
969  }
970  rb_print_undef(klass, id, 0);
971  }

973  if (flag == NOEX_UNDEF) {
974  flag = me->flag;

976  const char *v = "";
977  switch (flag & NOEX_MASK) {
978  case NOEX_PRIVATE: v = "private"; break;
979  case NOEX_PROTECTED: v = "protected"; break;
980  }
981  rb_name_error(id, "method `%s' for %s `%s' is %s",
982  rb_id2name(id),
983  (RB_TYPE_P(klass, T_MODULE)) ? "module" : "class",
984  rb_class2name(klass),
985  v);
986  }
987  }
988  if (def && def->type == VM_METHOD_TYPE_ZSUPER) {
989  klass = RCLASS_SUPER(defined_class);
990  id = def->original_id;
991  goto again;
992  }
993 
994  klass = defined_class;
995 
996  while (rclass != klass &&
997  (FL_TEST(rclass, FL_SINGLETON) || RB_TYPE_P(rclass, T_ICLASS))) {
998  rclass = RCLASS_SUPER(rclass);
999  }
1000 
1001  if (RB_TYPE_P(klass, T_ICLASS)) {
1002  klass = RBASIC(klass)->klass;
1003  }
1004 
1005  gen_method:
1006  method = TypedData_Make_Struct(mclass, struct METHOD, &method_data_type, data);
1007 
1008  data->recv = obj;
1009  data->rclass = rclass;
1010  data->defined_class = defined_class;
1011  data->id = rid;
1012  data->me = ALLOC(rb_method_entry_t);
1013  *data->me = *me;
1014  data->me->def->alias_count++;
1015  data->ume = ALLOC(struct unlinked_method_entry_list_entry);
1016

1018

1024  *
1025  * Document-class : Method
1026  *
1027  * Method objects are created by <code>Object#method</code>, and are
1028  * associated with a particular object (not just with a class). They
1029  * may be used to invoke the method within the object, and as a block
1030  * associated with an iterator. They may also be unbound from one
1031  * object (creating an <code>UnboundMethod</code>) and bound to
1032  * another.
1033  *
1034  * class Thing
1035  * def square(n)
1036  * n*n
1037  * end
1038  * end
1039  * thing = Thing.new
1040  * meth = thing.method(:square)
1041  *
1042  * meth.call(9) #=> 81
1043  * [ 1, 2, 3 ].collect(&meth) #=> [1, 4, 9]
1044  *
1045  */

1048  * call-seq:
1049  * meth == other_meth -> true or false
1050  *
1051  * Two method objects are equal if they are bound to the same
1052  * object and refer to the same method definition and their owners are the
1053  * same class or module.
1054  */

1057 method_eq(VALUE method, VALUE other)
1058 {
1059  struct METHOD *m1, *m2;
1060 
1061  if (!rb_obj_is_method(other))
1062  return Qfalse;
1063  if (CLASS_OF(method) != CLASS_OF(other))
1064  return Qfalse;

1066  Check_TypedStruct(method, &method_data_type);
1067  m1 = (struct METHOD *)DATA_PTR(method);
1068  m2 = (struct METHOD *)DATA_PTR(other);

1071  m1->rclass != m2->rclass ||

1073  return Qfalse;
1074  }
1075

1077 }

1355 {

1357  VALUE body;

1359 
1360  if (argc == 1) {
1361  id = rb_to_id(argv[0]);
1362  body = rb_block_lambda();
1363  }
1364  else {
1365  rb_check_arity(argc, 1, 2);
1366  id = rb_to_id(argv[0]);
1367  body = argv[1];

1369  rb_raise(rb_eTypeError,
1370  "wrong argument type %s (expected Proc/Method)",
1371  rb_obj_classname(body));
1372  }
1373  }
1374 
1375  if (rb_obj_is_method(body)) {
1376  struct METHOD *method = (struct METHOD *)DATA_PTR(body);
1377  VALUE rclass = method->rclass;
1378  if (rclass != mod && !RB_TYPE_P(rclass, T_MODULE) &&
1379  !RTEST(rb_class_inherited_p(mod, rclass))) {
1380  if (FL_TEST(rclass, FL_SINGLETON)) {
1381  rb_raise(rb_eTypeError,
1382  "can't bind singleton method to a different class");
1383  }
1384  else {
1385  rb_raise(rb_eTypeError,
1386  "bind argument must be a subclass of %s",
1387  rb_class2name(rclass));
1388  }
1389  }
1390  rb_method_entry_set(mod, id, method->me, noex);
1391  }
1392  else if (rb_obj_is_proc(body)) {
1393  rb_proc_t *proc;
1394  body = proc_dup(body);
1395  GetProcPtr(body, proc);
1396  if (BUILTIN_TYPE(proc->block.iseq) != T_NODE) {
1397  proc->block.iseq->defined_method_id = id;
1398  proc->block.iseq->klass = mod;
1399  proc->is_lambda = TRUE;
1400  proc->is_from_method = TRUE;
1401  proc->block.klass = mod;
1402  }

1404  }
1405  else {

1407  rb_raise(rb_eTypeError, "wrong argument type (expected Proc/Method)");
1408  }

1567 /**********************************************************************
1568  *
1569  * Document-class: UnboundMethod
1570  *
1571  * Ruby supports two forms of objectified methods. Class
1572  * <code>Method</code> is used to represent methods that are associated
1573  * with a particular object: these method objects are bound to that
1574  * object. Bound method objects for an object can be created using
1575  * <code>Object#method</code>.
1576  *
1577  * Ruby also supports unbound methods; methods objects that are not
1578  * associated with a particular object. These can be created either by
1579  * calling <code>Module#instance_method</code> or by calling
1580  * <code>unbind</code> on a bound method object. The result of both of
1581  * these is an <code>UnboundMethod</code> object.
1582  *
1583  * Unbound methods can only be called after they are bound to an
1584  * object. That object must be be a kind_of? the method's original
1585  * class.
1586  *
1587  * class Square
1588  * def area
1589  * @side * @side
1590  * end
1591  * def initialize(side)
1592  * @side = side
1593  * end
1594  * end
1595  *
1596  * area_un = Square.instance_method(:area)
1597  *
1598  * s = Square.new(12)
1599  * area = area_un.bind(s)
1600  * area.call #=> 144
1601  *
1602  * Unbound methods are a reference to the method at the time it was
1603  * objectified: subsequent changes to the underlying class will not
1604  * affect the unbound method.
1605  *
1606  * class Test
1607  * def test
1608  * :original
1609  * end
1610  * end
1611  * um = Test.instance_method(:test)
1612  * class Test
1613  * def test
1614  * :modified
1615  * end
1616  * end
1617  * t = Test.new
1618  * t.test #=> :modified
1619  * um.bind(t).call #=> :original
1620  *
1621  */
1622

1624  * call-seq:
1625  * umeth.bind(obj) -> method
1626  *
1627  * Bind <i>umeth</i> to <i>obj</i>. If <code>Klass</code> was the class
1628  * from which <i>umeth</i> was obtained,
1629  * <code>obj.kind_of?(Klass)</code> must be true.
1630  *
1631  * class A
1632  * def test
1633  * puts "In test, class = #{self.class}"
1634  * end

1636  * class B < A
1637  * end
1638  * class C < B

1642  * um = B.instance_method(:test)
1643  * bm = um.bind(C.new)
1644  * bm.call
1645  * bm = um.bind(B.new)
1646  * bm.call

1648  * bm.call

1651  *

1689 /*
1690  * Returns the number of required parameters and stores the maximum
1691  * number of parameters in max, or UNLIMITED_ARGUMENTS
1692  * if there is no maximum.
1693  */
1694 static int
1695 rb_method_entry_min_max_arity(const rb_method_entry_t *me, int *max)
1696 {

1698  if (!def) return *max = 0;

1700  case VM_METHOD_TYPE_CFUNC:
1701  if (def->body.cfunc.argc < 0) {
1702  *max = UNLIMITED_ARGUMENTS;
1703  return 0;
1704  }
1705  return *max = check_argc(def->body.cfunc.argc);
1706  case VM_METHOD_TYPE_ZSUPER:
1707  *max = UNLIMITED_ARGUMENTS;
1708  return 0;
1709  case VM_METHOD_TYPE_ATTRSET:
1710  return *max = 1;
1711  case VM_METHOD_TYPE_IVAR:

1717  return rb_iseq_min_max_arity(iseq, max);
1718  }
1719  case VM_METHOD_TYPE_UNDEF:
1720  case VM_METHOD_TYPE_NOTIMPLEMENTED:
1721  return *max = 0;
1722  case VM_METHOD_TYPE_MISSING:
1723  *max = UNLIMITED_ARGUMENTS;
1724  return 0;
1725  case VM_METHOD_TYPE_OPTIMIZED: {
1726  switch (def->body.optimize_type) {
1727  case OPTIMIZED_METHOD_TYPE_SEND:
1728  *max = UNLIMITED_ARGUMENTS;
1729  return 0;
1730  default:
1731  break;
1732  }
1733  }
1734  case VM_METHOD_TYPE_REFINED:
1735  *max = UNLIMITED_ARGUMENTS;
1736  return 0;
1737  }

1740 }
