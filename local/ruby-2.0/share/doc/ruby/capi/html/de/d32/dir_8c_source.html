101 static char *
102 bracket(
103  const char *p, /* pattern (next to '[') */
104  const char *pend,
105  const char *s, /* string */
106  const char *send,
107  int flags,
108  rb_encoding *enc)
109 {
110  const int nocase = flags & FNM_CASEFOLD;
111  const int escape = !(flags & FNM_NOESCAPE);
112  unsigned int c1, c2;
113  int r;
114  int ok = 0, not = 0;
115 
116  if (p >= pend) return NULL;
117  if (*p == '!' || *p == '^') {
118  not = 1;
119  p++;
120  }
121 
122  while (*p != ']') {
123  const char *t1 = p;
124  if (escape && *t1 == '\\')
125  t1++;
126  if (!*t1)
127  return NULL;
128  p = t1 + (r = rb_enc_mbclen(t1, pend, enc));
129  if (p >= pend) return NULL;
130  if (p[0] == '-' && p[1] != ']') {
131  const char *t2 = p + 1;
132  int r2;
133  if (escape && *t2 == '\\')
134  t2++;
135  if (!*t2)
136  return NULL;
137  p = t2 + (r2 = rb_enc_mbclen(t2, pend, enc));
138  if (ok) continue;
139  if ((r <= (send-s) && memcmp(t1, s, r) == 0) ||
140  (r2 <= (send-s) && memcmp(t2, s, r2) == 0)) {
141  ok = 1;
142  continue;
143  }
144  c1 = rb_enc_codepoint(s, send, enc);
145  if (nocase) c1 = rb_enc_toupper(c1, enc);
146  c2 = rb_enc_codepoint(t1, pend, enc);
147  if (nocase) c2 = rb_enc_toupper(c2, enc);
148  if (c1 < c2) continue;
149  c2 = rb_enc_codepoint(t2, pend, enc);
150  if (nocase) c2 = rb_enc_toupper(c2, enc);
151  if (c1 > c2) continue;
152  }
153  else {
154  if (ok) continue;
155  if (r <= (send-s) && memcmp(t1, s, r) == 0) {
156  ok = 1;
157  continue;
158  }
159  if (!nocase) continue;
160  c1 = rb_enc_toupper(rb_enc_codepoint(s, send, enc), enc);
161  c2 = rb_enc_toupper(rb_enc_codepoint(p, pend, enc), enc);
162  if (c1 != c2) continue;
163  }
164  ok = 1;
165  }
166 
167  return ok == not ? NULL : (char *)p + 1;
168 }

179 static int
180 fnmatch_helper(
181  const char **pcur, /* pattern */
182  const char **scur, /* string */
183  int flags,
184  rb_encoding *enc)
185 {
186  const int period = !(flags & FNM_DOTMATCH);
187  const int pathname = flags & FNM_PATHNAME;
188  const int escape = !(flags & FNM_NOESCAPE);
189  const int nocase = flags & FNM_CASEFOLD;
190 
191  const char *ptmp = 0;
192  const char *stmp = 0;
193 
194  const char *p = *pcur;
195  const char *pend = p + strlen(p);
196  const char *s = *scur;
197  const char *send = s + strlen(s);
198 
199  int r;
200 
201  if (period && *s == '.' && *UNESCAPE(p) != '.') /* leading period */
202  RETURN(FNM_NOMATCH);
203 
204  while (1) {
205  switch (*p) {
206  case '*':
207  do { p++; } while (*p == '*');
208  if (ISEND(UNESCAPE(p))) {
209  p = UNESCAPE(p);
210  RETURN(0);
211  }
212  if (ISEND(s))
213  RETURN(FNM_NOMATCH);
214  ptmp = p;
215  stmp = s;
216  continue;
217 
218  case '?':
219  if (ISEND(s))
220  RETURN(FNM_NOMATCH);
221  p++;
222  Inc(s, send, enc);
223  continue;
224 
225  case '[': {
226  const char *t;
227  if (ISEND(s))
228  RETURN(FNM_NOMATCH);
229  if ((t = bracket(p + 1, pend, s, send, flags, enc)) != 0) {
230  p = t;
231  Inc(s, send, enc);
232  continue;
233  }
234  goto failed;
235  }
236  }
237 
238  /* ordinary */
239  p = UNESCAPE(p);
240  if (ISEND(s))
241  RETURN(ISEND(p) ? 0 : FNM_NOMATCH);
242  if (ISEND(p))
243  goto failed;
244  r = rb_enc_precise_mbclen(p, pend, enc);
245  if (!MBCLEN_CHARFOUND_P(r))
246  goto failed;
247  if (r <= (send-s) && memcmp(p, s, r) == 0) {
248  p += r;
249  s += r;
250  continue;
251  }
252  if (!nocase) goto failed;
253  if (rb_enc_toupper(rb_enc_codepoint(p, pend, enc), enc) !=
254  rb_enc_toupper(rb_enc_codepoint(s, send, enc), enc))
255  goto failed;
256  p += r;
257  Inc(s, send, enc);
258  continue;
259 
260  failed: /* try next '*' position */
261  if (ptmp && stmp) {
262  p = ptmp;
263  Inc(stmp, send, enc); /* !ISEND(*stmp) */
264  s = stmp;
265  continue;
266  }
267  RETURN(FNM_NOMATCH);
268  }
269 }

271 static int
272 fnmatch(
273  const char *pattern,
274  rb_encoding *enc,
275  const char *string,
276  int flags)
277 {
278  const char *p = pattern;
279  const char *s = string;
280  const char *send = s + strlen(string);
281  const int period = !(flags & FNM_DOTMATCH);
282  const int pathname = flags & FNM_PATHNAME;
283 
284  const char *ptmp = 0;
285  const char *stmp = 0;
286 
287  if (pathname) {
288  while (1) {
289  if (p[0] == '*' && p[1] == '*' && p[2] == '/') {
290  do { p += 3; } while (p[0] == '*' && p[1] == '*' && p[2] == '/');
291  ptmp = p;
292  stmp = s;
293  }
294  if (fnmatch_helper(&p, &s, flags, enc) == 0) {
295  while (*s && *s != '/') Inc(s, send, enc);
296  if (*p && *s) {
297  p++;
298  s++;
299  continue;
300  }
301  if (!*p && !*s)
302  return 0;
303  }
304  /* failed : try next recursion */
305  if (ptmp && stmp && !(period && *stmp == '.')) {
306  while (*stmp && *stmp != '/') Inc(stmp, send, enc);
307  if (*stmp) {
308  p = ptmp;
309  stmp++;
310  s = stmp;
311  continue;
312  }
313  }
314  return FNM_NOMATCH;
315  }
316  }
317  else
318  return fnmatch_helper(&p, &s, flags, enc);
319 }

833  *
834  * /var/spool/mail
835  * /tmp
836  * /usr

1177 {

1181 
1182  while (p < e && *p) {

1184  if (!tmp) goto error;

1186  /* fold continuous RECURSIVEs (needed in glob_helper) */
1187  do { p += 3; while (*p == '/') p++; } while (p[0] == '*' && p[1] == '*' && p[2] == '/');
1188  tmp->type = RECURSIVE;
1189  tmp->str = 0;
1190  dirsep = 1;
1191  recursive = 1;
1192  }
1193  else {
1194  const char *m = find_dirsep(p, e, flags, enc);
1195  int magic = has_magic(p, m, flags, enc);
1196  char *buf;
1197 
1198  if (!magic && !recursive && *m) {
1199  const char *m2;
1200  while (!has_magic(m+1, m2 = find_dirsep(m+1, e, flags, enc), flags, enc) &&
1201  *m2) {
1202  m = m2;
1203  }
1204  }
1205  buf = GLOB_ALLOC_N(char, m-p+1);
1206  if (!buf) {
1207  GLOB_FREE(tmp);
1208  goto error;
1209  }
1210  memcpy(buf, p, m-p);
1211  buf[m-p] = '\0';

1213  tmp->str = buf;
1214  if (*m) {
1215  dirsep = 1;
1216  p = m + 1;
1217  }
1218  else {
1219  dirsep = 0;
1220  p = m;
1221  }
1222  }

1225  }

1231  glob_free_pattern(list);

1303 glob_helper(
1304  const char *path,
1305  int dirsep, /* '/' should be placed before appending child entry's name to 'path'. */
1306  enum answer exist, /* Does 'path' indicate an existing entry? */
1307  enum answer isdir, /* Does 'path' indicate a directory or a symlink to a directory? */
1308  struct glob_pattern **beg,
1309  struct glob_pattern **end,
1310  int flags,
1311  ruby_glob_func *func,
1312  VALUE arg,
1313  rb_encoding *enc)
1314 {

1316  int status = 0;
1317  struct glob_pattern **cur, **new_beg, **new_end;
1318  int plain = 0, magical = 0, recursive = 0, match_all = 0, match_dir = 0;
1319  int escape = !(flags & FNM_NOESCAPE);
1320 
1321  for (cur = beg; cur < end; ++cur) {
1322  struct glob_pattern *p = *cur;
1323  if (p->type == RECURSIVE) {
1324  recursive = 1;
1325  p = p->next;
1326  }
1327  switch (p->type) {
1328  case PLAIN:
1329  plain = 1;
1330  break;
1331  case MAGICAL:
1332  magical = 1;
1333  break;
1334  case MATCH_ALL:
1335  match_all = 1;
1336  break;
1337  case MATCH_DIR:
1338  match_dir = 1;
1339  break;
1340  case RECURSIVE:
1341  rb_bug("continuous RECURSIVEs");
1342  }
1343  }
1344 
1345  if (*path) {
1346  if (match_all && exist == UNKNOWN) {
1347  if (do_lstat(path, &st, flags) == 0) {
1348  exist = YES;
1349  isdir = S_ISDIR(st.st_mode) ? YES : S_ISLNK(st.st_mode) ? UNKNOWN : NO;
1350  }
1351  else {
1352  exist = NO;
1353  isdir = NO;
1354  }
1355  }
1356  if (match_dir && isdir == UNKNOWN) {
1357  if (do_stat(path, &st, flags) == 0) {
1358  exist = YES;
1359  isdir = S_ISDIR(st.st_mode) ? YES : NO;
1360  }
1361  else {
1362  exist = NO;
1363  isdir = NO;
1364  }
1365  }
1366  if (match_all && exist == YES) {
1367  status = glob_call_func(func, path, arg, enc);
1368  if (status) return status;
1369  }
1370  if (match_dir && isdir == YES) {
1371  char *tmp = join_path(path, dirsep, "", 0);
1372  if (!tmp) return -1;
1373  status = glob_call_func(func, tmp, arg, enc);
1374  GLOB_FREE(tmp);
1375  if (status) return status;
1376  }
1377  }
1378 
1379  if (exist == NO || isdir == NO) return 0;
1380 
1381  if (magical || recursive) {
1382  struct dirent *dp;
1383  DIR *dirp;
1384  IF_HAVE_READDIR_R(DEFINE_STRUCT_DIRENT entry);
1385  dirp = do_opendir(*path ? path : ".", flags, enc);
1386  if (dirp == NULL) return 0;
1387 
1388  while (READDIR(dirp, enc, &STRUCT_DIRENT(entry), dp)) {
1389  char *buf;
1390  enum answer new_isdir = UNKNOWN;
1391 
1392  if (recursive && dp->d_name[0] == '.') {
1393  /* always skip current and parent directories not to recurse infinitely */
1394  if (!dp->d_name[1]) continue;
1395  if (dp->d_name[1] == '.' && !dp->d_name[2]) continue;
1396  }
1397

1399  if (!buf) {
1400  status = -1;
1401  break;
1402  }
1403  if (recursive && ((flags & FNM_DOTMATCH) || dp->d_name[0] != '.')) {
1404  /* RECURSIVE never match dot files unless FNM_DOTMATCH is set */

1407  new_isdir = S_ISDIR(st.st_mode) ? YES : S_ISLNK(st.st_mode) ? UNKNOWN : NO;
1408  else

1416  if (!new_beg) {
1417  GLOB_FREE(buf);
1418  status = -1;
1419  break;
1420  }
1421 
1422  for (cur = beg; cur < end; ++cur) {
1423  struct glob_pattern *p = *cur;

1427  p = p->next; /* 0 times recursion */
1428  }

1431  *new_end++ = p->next;
1432  }
1433  }

1438  GLOB_FREE(new_beg);
1439  if (status) break;
1440  }
1441

1443  }
1444  else if (plain) {

1446

1450  *copy_end++ = (*cur)->type == PLAIN ? *cur : 0;
1451 
1452  for (cur = copy_beg; cur < copy_end; ++cur) {
1453  if (*cur) {
1454  char *buf;
1455  char *name;

1457  name = GLOB_ALLOC_N(char, len);
1458  if (!name) {
1459  status = -1;
1460  break;
1461  }
1462  memcpy(name, (*cur)->str, len);
1463  if (escape)
1464  len = remove_backslashes(name, name+len-1, enc) - name;
1465 
1466  new_beg = new_end = GLOB_ALLOC_N(struct glob_pattern *, end - beg);
1467  if (!new_beg) {
1468  GLOB_FREE(name);
1469  status = -1;
1470  break;
1471  }
1472  *new_end++ = (*cur)->next;
1473  for (cur2 = cur + 1; cur2 < copy_end; ++cur2) {
1474  if (*cur2 && fnmatch((*cur2)->str, enc, name, flags) == 0) {
1475  *new_end++ = (*cur2)->next;
1476  *cur2 = 0;
1477  }
1478  }
1479 
1480  buf = join_path(path, dirsep, name, len);
1481  GLOB_FREE(name);
1482  if (!buf) {
1483  GLOB_FREE(new_beg);
1484  status = -1;
1485  break;
1486  }
1487  status = glob_helper(buf, 1, UNKNOWN, UNKNOWN, new_beg,
1488  new_end, flags, func, arg, enc);
1489  GLOB_FREE(buf);
1490  GLOB_FREE(new_beg);
1491  if (status) break;
1492  }
1493  }
1494 
1495  GLOB_FREE(copy_beg);
1496  }
1497

1499 }

1586 static int
1587 ruby_brace_expand(const char *str, int flags, ruby_glob_func *func, VALUE arg,
1588  rb_encoding *enc)
1589 {
1590  const int escape = !(flags & FNM_NOESCAPE);
1591  const char *p = str;
1592  const char *pend = p + strlen(p);
1593  const char *s = p;
1594  const char *lbrace = 0, *rbrace = 0;
1595  int nest = 0, status = 0;
1596 
1597  while (*p) {
1598  if (*p == '{' && nest++ == 0) {
1599  lbrace = p;
1600  }
1601  if (*p == '}' && --nest <= 0) {
1602  rbrace = p;
1603  break;
1604  }
1605  if (*p == '\\' && escape) {
1606  if (!*++p) break;
1607  }
1608  Inc(p, pend, enc);
1609  }
1610 
1611  if (lbrace && rbrace) {
1612  size_t len = strlen(s) + 1;
1613  char *buf = GLOB_ALLOC_N(char, len);
1614  long shift;
1615 
1616  if (!buf) return -1;
1617  memcpy(buf, s, lbrace-s);
1618  shift = (lbrace-s);
1619  p = lbrace;
1620  while (p < rbrace) {
1621  const char *t = ++p;
1622  nest = 0;
1623  while (p < rbrace && !(*p == ',' && nest == 0)) {
1624  if (*p == '{') nest++;
1625  if (*p == '}') nest--;
1626  if (*p == '\\' && escape) {
1627  if (++p == rbrace) break;
1628  }
1629  Inc(p, pend, enc);
1630  }
1631  memcpy(buf+shift, t, p-t);
1632  strlcpy(buf+shift+(p-t), rbrace+1, len-(shift+(p-t)));
1633  status = ruby_brace_expand(buf, flags, func, arg, enc);
1634  if (status) break;
1635  }
1636  GLOB_FREE(buf);
1637  }
1638  else if (!lbrace && !rbrace) {
1639  status = (*func)(s, arg, enc);
1640  }
1641 
1642  return status;
1643 }

1763 /*
1764  * call-seq:
1765  * Dir.glob( pattern, [flags] ) -> array
1766  * Dir.glob( pattern, [flags] ) {| filename | block } -> nil
1767  *
1768  * Returns the filenames found by expanding <i>pattern</i> which is
1769  * an +Array+ of the patterns or the pattern +String+, either as an
1770  * <i>array</i> or as parameters to the block. Note that this pattern
1771  * is not a regexp (it's closer to a shell glob). See
1772  * <code>File::fnmatch</code> for the meaning of the <i>flags</i>
1773  * parameter. Note that case sensitivity depends on your system (so
1774  * <code>File::FNM_CASEFOLD</code> is ignored), as does the order
1775  * in which the results are returned.
1776  *
1777  * <code>*</code>:: Matches any file. Can be restricted by
1778  * other values in the glob. <code>*</code>
1779  * will match all files; <code>c*</code> will
1780  * match all files beginning with
1781  * <code>c</code>; <code>*c</code> will match
1782  * all files ending with <code>c</code>; and
1783  * <code>\*c\*</code> will match all files that
1784  * have <code>c</code> in them (including at
1785  * the beginning or end). Equivalent to
1786  * <code>/ .* /x</code> in regexp. Note, this
1787  * will not match Unix-like hidden files (dotfiles).
1788  * In order to include those in the match results,
1789  * you must use something like <code>"{*,.*}"</code>.
1790  * <code>**</code>:: Matches directories recursively.
1791  * <code>?</code>:: Matches any one character. Equivalent to
1792  * <code>/.{1}/</code> in regexp.
1793  * <code>[set]</code>:: Matches any one character in +set+.
1794  * Behaves exactly like character sets in
1795  * Regexp, including set negation
1796  * (<code>[^a-z]</code>).
1797  * <code>{p,q}</code>:: Matches either literal <code>p</code> or
1798  * literal <code>q</code>. Matching literals
1799  * may be more than one character in length.
1800  * More than two literals may be specified.
1801  * Equivalent to pattern alternation in
1802  * regexp.
1803  * <code> \\ </code>:: Escapes the next metacharacter.
1804  * Note that this means you cannot use backslash
1805  * in windows as part of a glob,
1806  * i.e. <code>Dir["c:\\foo*"]</code> will not work,
1807  * use <code>Dir["c:/foo*"]</code> instead.
1808  *
1809  * Dir["config.?"] #=> ["config.h"]
1810  * Dir.glob("config.?") #=> ["config.h"]
1811  * Dir.glob("*.[a-z][a-z]") #=> ["main.rb"]
1812  * Dir.glob("*.[^r]*") #=> ["config.h"]
1813  * Dir.glob("*.{rb,h}") #=> ["main.rb", "config.h"]
1814  * Dir.glob("*") #=> ["config.h", "main.rb"]
1815  * Dir.glob("*", File::FNM_DOTMATCH) #=> [".", "..", "config.h", "main.rb"]
1816  *
1817  * rbfiles = File.join("**", "*.rb")
1818  * Dir.glob(rbfiles) #=> ["main.rb",
1819  * # "lib/song.rb",
1820  * # "lib/song/karaoke.rb"]
1821  * libdirs = File.join("**", "lib")
1822  * Dir.glob(libdirs) #=> ["lib"]
1823  *
1824  * librbfiles = File.join("**", "lib", "**", "*.rb")
1825  * Dir.glob(librbfiles) #=> ["lib/song.rb",
1826  * # "lib/song/karaoke.rb"]
1827  *
1828  * librbfiles = File.join("**", "lib", "*.rb")
1829  * Dir.glob(librbfiles) #=> ["lib/song.rb"]
1830  */

1945 /*
1946  * call-seq:
1947  * File.fnmatch( pattern, path, [flags] ) -> (true or false)
1948  * File.fnmatch?( pattern, path, [flags] ) -> (true or false)
1949  *
1950  * Returns true if <i>path</i> matches against <i>pattern</i> The
1951  * pattern is not a regular expression; instead it follows rules
1952  * similar to shell filename globbing. It may contain the following
1953  * metacharacters:
1954  *
1955  * <code>*</code>:: Matches any file. Can be restricted by
1956  * other values in the glob. <code>*</code>
1957  * will match all files; <code>c*</code> will
1958  * match all files beginning with
1959  * <code>c</code>; <code>*c</code> will match
1960  * all files ending with <code>c</code>; and
1961  * <code>\*c*</code> will match all files that
1962  * have <code>c</code> in them (including at
1963  * the beginning or end). Equivalent to
1964  * <code>/ .* /x</code> in regexp.
1965  * <code>**</code>:: Matches directories recursively or files
1966  * expansively.
1967  * <code>?</code>:: Matches any one character. Equivalent to
1968  * <code>/.{1}/</code> in regexp.
1969  * <code>[set]</code>:: Matches any one character in +set+.
1970  * Behaves exactly like character sets in
1971  * Regexp, including set negation
1972  * (<code>[^a-z]</code>).
1973  * <code> \ </code>:: Escapes the next metacharacter.
1974  *
1975  * <i>flags</i> is a bitwise OR of the <code>FNM_xxx</code>
1976  * parameters. The same glob pattern and flags are used by
1977  * <code>Dir::glob</code>.
1978  *
1979  * File.fnmatch('cat', 'cat') #=> true # match entire string
1980  * File.fnmatch('cat', 'category') #=> false # only match partial string
1981  * File.fnmatch('c{at,ub}s', 'cats') #=> false # { } isn't supported
1982  *
1983  * File.fnmatch('c?t', 'cat') #=> true # '?' match only 1 character
1984  * File.fnmatch('c??t', 'cat') #=> false # ditto
1985  * File.fnmatch('c*', 'cats') #=> true # '*' match 0 or more characters
1986  * File.fnmatch('c*t', 'c/a/b/t') #=> true # ditto
1987  * File.fnmatch('ca[a-z]', 'cat') #=> true # inclusive bracket expression
1988  * File.fnmatch('ca[^t]', 'cat') #=> false # exclusive bracket expression ('^' or '!')
1989  *
1990  * File.fnmatch('cat', 'CAT') #=> false # case sensitive
1991  * File.fnmatch('cat', 'CAT', File::FNM_CASEFOLD) #=> true # case insensitive
1992  *
1993  * File.fnmatch('?', '/', File::FNM_PATHNAME) #=> false # wildcard doesn't match '/' on FNM_PATHNAME
1994  * File.fnmatch('*', '/', File::FNM_PATHNAME) #=> false # ditto
1995  * File.fnmatch('[/]', '/', File::FNM_PATHNAME) #=> false # ditto
1996  *
1997  * File.fnmatch('\?', '?') #=> true # escaped wildcard becomes ordinary
1998  * File.fnmatch('\a', 'a') #=> true # escaped ordinary remains ordinary
1999  * File.fnmatch('\a', '\a', File::FNM_NOESCAPE) #=> true # FNM_NOESCAPE makes '\' ordinary
2000  * File.fnmatch('[\?]', '?') #=> true # can escape inside bracket expression
2001  *
2002  * File.fnmatch('*', '.profile') #=> false # wildcard doesn't match leading
2003  * File.fnmatch('*', '.profile', File::FNM_DOTMATCH) #=> true # period by default.
2004  * File.fnmatch('.*', '.profile') #=> true
2005  *
2006  * rbfiles = '**' '/' '*.rb' # you don't have to do like this. just write in single string.
2007  * File.fnmatch(rbfiles, 'main.rb') #=> false
2008  * File.fnmatch(rbfiles, './main.rb') #=> false
