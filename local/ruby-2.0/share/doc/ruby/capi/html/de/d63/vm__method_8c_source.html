224 rb_method_entry_make(VALUE klass, ID mid, rb_method_type_t type,
225  rb_method_definition_t *def, rb_method_flag_t noex)
226 {
227  rb_method_entry_t *me;
228 #if NOEX_NOREDEF
229  VALUE rklass;
230 #endif
231  st_table *mtbl;
232  st_data_t data;
233  int make_refined = 0;
234 
235  if (NIL_P(klass)) {
236  klass = rb_cObject;
237  }
238  if (rb_safe_level() >= 4 &&
239  (klass == rb_cObject || !OBJ_UNTRUSTED(klass))) {
240  rb_raise(rb_eSecurityError, "Insecure: can't define method");
241  }
242  if (!FL_TEST(klass, FL_SINGLETON) &&
243  type != VM_METHOD_TYPE_NOTIMPLEMENTED &&
244  type != VM_METHOD_TYPE_ZSUPER &&
245  (mid == idInitialize || mid == idInitialize_copy ||
246  mid == idInitialize_clone || mid == idInitialize_dup ||
247  mid == idRespond_to_missing)) {
248  noex = NOEX_PRIVATE | noex;
249  }
250 
251  rb_check_frozen(klass);
252 #if NOEX_NOREDEF
253  rklass = klass;
254 #endif
255  if (FL_TEST(klass, RMODULE_IS_REFINEMENT)) {
256  VALUE refined_class =
257  rb_refinement_module_get_refined_class(klass);
258 
259  rb_add_refined_method_entry(refined_class, mid);
260  }
261  if (type == VM_METHOD_TYPE_REFINED) {
262  rb_method_entry_t *old_me =
263  lookup_method_table(RCLASS_ORIGIN(klass), mid);
264  if (old_me) rb_vm_check_redefinition_opt_method(old_me, klass);
265  }
266  else {
267  klass = RCLASS_ORIGIN(klass);
268  }
269  mtbl = RCLASS_M_TBL(klass);
270 
271  /* check re-definition */
272  if (st_lookup(mtbl, mid, &data)) {
273  rb_method_entry_t *old_me = (rb_method_entry_t *)data;
274  rb_method_definition_t *old_def = old_me->def;
275 
276  if (rb_method_definition_eq(old_def, def)) return old_me;
277 #if NOEX_NOREDEF
278  if (old_me->flag & NOEX_NOREDEF) {
279  rb_raise(rb_eTypeError, "cannot redefine %"PRIsVALUE"#%"PRIsVALUE,
280  rb_class_name(rklass), rb_id2str(mid));
281  }
282 #endif
283  rb_vm_check_redefinition_opt_method(old_me, klass);
284  if (old_def->type == VM_METHOD_TYPE_REFINED)
285  make_refined = 1;
286 
287  if (RTEST(ruby_verbose) &&
288  type != VM_METHOD_TYPE_UNDEF &&
289  old_def->alias_count == 0 &&
290  old_def->type != VM_METHOD_TYPE_UNDEF &&
291  old_def->type != VM_METHOD_TYPE_ZSUPER) {
292  rb_iseq_t *iseq = 0;
293 
294  rb_warning("method redefined; discarding old %s", rb_id2name(mid));
295  switch (old_def->type) {
296  case VM_METHOD_TYPE_ISEQ:
297  iseq = old_def->body.iseq;
298  break;
299  case VM_METHOD_TYPE_BMETHOD:
300  iseq = rb_proc_get_iseq(old_def->body.proc, 0);
301  break;
302  default:
303  break;
304  }
305  if (iseq && !NIL_P(iseq->location.path)) {
306  int line = iseq->line_info_table ? rb_iseq_first_lineno(iseq) : 0;
307  rb_compile_warning(RSTRING_PTR(iseq->location.path), line,
308  "previous definition of %s was here",
309  rb_id2name(old_def->original_id));
310  }
311  }
312 
313  rb_unlink_method_entry(old_me);
314  }
315 
316  me = ALLOC(rb_method_entry_t);
317 
318  rb_clear_cache_by_id(mid);
319 
320  me->flag = NOEX_WITH_SAFE(noex);
321  me->mark = 0;
322  me->called_id = mid;
323  me->klass = klass;
324  me->def = def;
325  if (def) def->alias_count++;
326 
327  /* check mid */
328  if (klass == rb_cObject && mid == idInitialize) {
329  rb_warn("redefining Object#initialize may cause infinite loop");
330  }
331  /* check mid */
332  if (mid == object_id || mid == id__send__) {
333  if (type == VM_METHOD_TYPE_ISEQ) {
334  rb_warn("redefining `%s' may cause serious problems", rb_id2name(mid));
335  }
336  }
337 
338  if (make_refined) {
339  make_method_entry_refined(me);
340  }
341 
342  st_insert(mtbl, mid, (st_data_t) me);
343 
344  return me;
345 }

401 rb_method_entry_t *
402 rb_add_method(VALUE klass, ID mid, rb_method_type_t type, void *opts, rb_method_flag_t noex)
403 {
404  rb_thread_t *th;
405  rb_control_frame_t *cfp;
406  int line;
407  rb_method_entry_t *me = rb_method_entry_make(klass, mid, type, 0, noex);
408  rb_method_definition_t *def = ALLOC(rb_method_definition_t);
409  if (me->def && me->def->type == VM_METHOD_TYPE_REFINED) {
410  me->def->body.orig_me->def = def;
411  }
412  else {
413  me->def = def;
414  }
415  def->type = type;
416  def->original_id = mid;
417  def->alias_count = 0;
418  switch (type) {
419  case VM_METHOD_TYPE_ISEQ:
420  def->body.iseq = (rb_iseq_t *)opts;
421  break;
422  case VM_METHOD_TYPE_CFUNC:
423  {
424  rb_method_cfunc_t *cfunc = (rb_method_cfunc_t *)opts;
425  setup_method_cfunc_struct(&def->body.cfunc, cfunc->func, cfunc->argc);
426  }
427  break;
428  case VM_METHOD_TYPE_ATTRSET:
429  case VM_METHOD_TYPE_IVAR:
430  def->body.attr.id = (ID)opts;
431  def->body.attr.location = Qfalse;
432  th = GET_THREAD();
433  cfp = rb_vm_get_ruby_level_next_cfp(th, th->cfp);
434  if (cfp && (line = rb_vm_get_sourceline(cfp))) {
435  VALUE location = rb_ary_new3(2, cfp->iseq->location.path, INT2FIX(line));
436  def->body.attr.location = rb_ary_freeze(location);
437  }
438  break;
439  case VM_METHOD_TYPE_BMETHOD:
440  def->body.proc = (VALUE)opts;
441  break;
442  case VM_METHOD_TYPE_NOTIMPLEMENTED:
443  setup_method_cfunc_struct(&def->body.cfunc, rb_f_notimplement, -1);
444  break;
445  case VM_METHOD_TYPE_OPTIMIZED:
446  def->body.optimize_type = (enum method_optimized_type)opts;
447  break;
448  case VM_METHOD_TYPE_ZSUPER:
449  case VM_METHOD_TYPE_UNDEF:
450  break;
451  case VM_METHOD_TYPE_REFINED:
452  def->body.orig_me = (rb_method_entry_t *) opts;
453  break;
454  default:
455  rb_bug("rb_add_method: unsupported method type (%d)\n", type);
456  }
457  if (type != VM_METHOD_TYPE_UNDEF && type != VM_METHOD_TYPE_REFINED) {
458  method_added(klass, mid);
459  }
460  return me;
461 }

650  rb_method_entry_t *me = 0;
651  VALUE self = klass;
652 
653  klass = RCLASS_ORIGIN(klass);
654  if (klass == rb_cObject) {

662  rb_warn("removing `%s' may cause serious problems", rb_id2name(mid));
663  }

666  !(me = (rb_method_entry_t *)data) ||
667  (!me->def || me->def->type == VM_METHOD_TYPE_UNDEF)) {
668  rb_name_error(mid, "method `%s' not defined in %s",
669  rb_id2name(mid), rb_class2name(klass));
670  }

672  st_delete(RCLASS_M_TBL(klass), &key, &data);
673 
674  rb_vm_check_redefinition_opt_method(me, klass);
675  rb_clear_cache_for_undef(klass, mid);
676  rb_unlink_method_entry(me);
677 
678  CALL_METHOD_HOOK(self, removed, mid);
679 }

735 {

768 int
769 rb_method_boundp(VALUE klass, ID id, int ex)
770 {
771  rb_method_entry_t *me =
772  rb_method_entry_without_refinements(klass, id, 0);
773 
774  if (me != 0) {
775  if ((ex & ~NOEX_RESPONDS) &&
776  ((me->flag & NOEX_PRIVATE) ||
777  ((ex & NOEX_RESPONDS) && (me->flag & NOEX_PROTECTED)))) {
778  return 0;
779  }
780  if (!me->def) return 0;
781  if (me->def->type == VM_METHOD_TYPE_NOTIMPLEMENTED) {
782  if (ex & NOEX_RESPONDS) return 2;
783  return 0;
784  }
785  return 1;
786  }
787  return 0;

790 void
791 rb_attr(VALUE klass, ID id, int read, int write, int ex)
792 {
793  const char *name;
794  ID attriv;
795  VALUE aname;
796  rb_method_flag_t noex;
797 
798  if (!ex) {
799  noex = NOEX_PUBLIC;
800  }
801  else {
802  if (SCOPE_TEST(NOEX_PRIVATE)) {
803  noex = NOEX_PRIVATE;
804  rb_warning((SCOPE_CHECK(NOEX_MODFUNC)) ?
805  "attribute accessor as module_function" :
806  "private attribute?");
807  }
808  else if (SCOPE_TEST(NOEX_PROTECTED)) {
809  noex = NOEX_PROTECTED;
810  }
811  else {

813  }

821  rb_raise(rb_eArgError, "argument needs to be symbol or string");
822  }
823  aname = rb_sprintf("@%s", name);
824  rb_enc_copy(aname, rb_id2str(id));
825  attriv = rb_intern_str(aname);
826  if (read) {
827  rb_add_method(klass, id, VM_METHOD_TYPE_IVAR, (void *)attriv, noex);
828  }
829  if (write) {
830  rb_add_method(klass, rb_id_attrset(id), VM_METHOD_TYPE_ATTRSET, (void *)attriv, noex);
831  }

834 void
835 rb_undef(VALUE klass, ID id)
836 {
837  rb_method_entry_t *me;
838 
839  if (NIL_P(klass)) {
840  rb_raise(rb_eTypeError, "no class to undef method");
841  }
842  if (rb_vm_cbase() == rb_cObject && klass == rb_cObject) {
843  rb_secure(4);
844  }
845  if (rb_safe_level() >= 4 && !OBJ_UNTRUSTED(klass)) {
846  rb_raise(rb_eSecurityError, "Insecure: can't undef `%s'", rb_id2name(id));
847  }
848  rb_frozen_class_p(klass);
849  if (id == object_id || id == id__send__ || id == idInitialize) {
850  rb_warn("undefining `%s' may cause serious problems", rb_id2name(id));
851  }
852 
853  me = search_method(klass, id, 0);
854 
855  if (UNDEFINED_METHOD_ENTRY_P(me)) {
856  const char *s0 = " class";
857  VALUE c = klass;
858 
859  if (FL_TEST(c, FL_SINGLETON)) {
860  VALUE obj = rb_ivar_get(klass, attached);
861 
862  if (RB_TYPE_P(obj, T_MODULE) || RB_TYPE_P(obj, T_CLASS)) {
863  c = obj;
864  s0 = "";
865  }
866  }
867  else if (RB_TYPE_P(c, T_MODULE)) {
868  s0 = " module";
869  }
870  rb_name_error(id, "undefined method `%"PRIsVALUE"' for%s `%"PRIsVALUE"'",
871  QUOTE_ID(id), s0, rb_class_name(c));
872  }
873 
874  rb_add_method(klass, id, VM_METHOD_TYPE_UNDEF, 0, NOEX_PUBLIC);
875 
876  CALL_METHOD_HOOK(klass, undefined, id);
877 }

879 /*
880  * call-seq:
881  * undef_method(symbol) -> self
882  *
883  * Prevents the current class from responding to calls to the named
884  * method. Contrast this with <code>remove_method</code>, which deletes
885  * the method from the particular class; Ruby will still search
886  * superclasses and mixed-in modules for a possible receiver.
887  *
888  * class Parent
889  * def hello
890  * puts "In parent"
891  * end
892  * end
893  * class Child < Parent
894  * def hello
895  * puts "In child"
896  * end
897  * end
898  *
899  *
900  * c = Child.new
901  * c.hello
902  *
903  *
904  * class Child
905  * remove_method :hello # remove from child, still in parent
906  * end
907  * c.hello
908  *
909  *
910  * class Child
911  * undef_method :hello # prevent any calls to 'hello'
912  * end
913  * c.hello
914  *
915  * <em>produces:</em>
916  *
917  * In child
918  * In parent
919  * prog.rb:23: undefined method `hello' for #<Child:0x401b3bb4> (NoMethodError)
920  */
921 
922 static VALUE
923 rb_mod_undef_method(int argc, VALUE *argv, VALUE mod)
924 {
925  int i;
926  for (i = 0; i < argc; i++) {
927  VALUE v = argv[i];
928  ID id = rb_check_id(&v);
929  if (!id) {
930  rb_method_name_error(mod, v);
931  }
932  rb_undef(mod, id);
933  }
934  return mod;
935 }

937 /*
938  * call-seq:
939  * mod.method_defined?(symbol) -> true or false
940  *
941  * Returns +true+ if the named method is defined by
942  * _mod_ (or its included modules and, if _mod_ is a class,
943  * its ancestors). Public and protected methods are matched.
944  *
945  * module A
946  * def method1() end
947  * end
948  * class B
949  * def method2() end
950  * end
951  * class C < B
952  * include A
953  * def method3() end
954  * end
955  *
956  * A.method_defined? :method1 #=> true
957  * C.method_defined? "method1" #=> true
958  * C.method_defined? "method2" #=> true
959  * C.method_defined? "method3" #=> true
960  * C.method_defined? "method4" #=> false
961  */

964 rb_mod_method_defined(VALUE mod, VALUE mid)
965 {

967  if (!id || !rb_method_boundp(mod, id, 1)) {
968  return Qfalse;
969  }
970  return Qtrue;
971 
972 }

1092 static int
1093 rb_method_definition_eq(const rb_method_definition_t *d1, const rb_method_definition_t *d2)
1094 {
1095  if (d1 && d1->type == VM_METHOD_TYPE_REFINED)
1096  d1 = d1->body.orig_me->def;
1097  if (d2 && d2->type == VM_METHOD_TYPE_REFINED)
1098  d2 = d2->body.orig_me->def;
1099  if (d1 == d2) return 1;
1100  if (!d1 || !d2) return 0;
1101  if (d1->type != d2->type) {
1102  return 0;
1103  }
1104  switch (d1->type) {
1105  case VM_METHOD_TYPE_ISEQ:
1106  return d1->body.iseq == d2->body.iseq;
1107  case VM_METHOD_TYPE_CFUNC:
1108  return
1109  d1->body.cfunc.func == d2->body.cfunc.func &&
1110  d1->body.cfunc.argc == d2->body.cfunc.argc;
1111  case VM_METHOD_TYPE_ATTRSET:
1112  case VM_METHOD_TYPE_IVAR:
1113  return d1->body.attr.id == d2->body.attr.id;
1114  case VM_METHOD_TYPE_BMETHOD:
1115  return RTEST(rb_equal(d1->body.proc, d2->body.proc));
1116  case VM_METHOD_TYPE_MISSING:
1117  return d1->original_id == d2->original_id;
1118  case VM_METHOD_TYPE_ZSUPER:
1119  case VM_METHOD_TYPE_NOTIMPLEMENTED:
1120  case VM_METHOD_TYPE_UNDEF:
1121  return 1;
1122  case VM_METHOD_TYPE_OPTIMIZED:
1123  return d1->body.optimize_type == d2->body.optimize_type;
1124  default:
1125  rb_bug("rb_method_entry_eq: unsupported method type (%d)\n", d1->type);
1126  return 0;
1127  }
1128 }

1130 static st_index_t
1131 rb_hash_method_definition(st_index_t hash, const rb_method_definition_t *def)
1132 {
1133  again:
1134  hash = rb_hash_uint(hash, def->type);
1135  switch (def->type) {
1136  case VM_METHOD_TYPE_ISEQ:
1137  return rb_hash_uint(hash, (st_index_t)def->body.iseq);
1138  case VM_METHOD_TYPE_CFUNC:
1139  hash = rb_hash_uint(hash, (st_index_t)def->body.cfunc.func);
1140  return rb_hash_uint(hash, def->body.cfunc.argc);
1141  case VM_METHOD_TYPE_ATTRSET:
1142  case VM_METHOD_TYPE_IVAR:
1143  return rb_hash_uint(hash, def->body.attr.id);
1144  case VM_METHOD_TYPE_BMETHOD:
1145  return rb_hash_proc(hash, def->body.proc);
1146  case VM_METHOD_TYPE_MISSING:
1147  return rb_hash_uint(hash, def->original_id);
1148  case VM_METHOD_TYPE_ZSUPER:
1149  case VM_METHOD_TYPE_NOTIMPLEMENTED:
1150  case VM_METHOD_TYPE_UNDEF:
1151  return hash;
1152  case VM_METHOD_TYPE_OPTIMIZED:
1153  return rb_hash_uint(hash, def->body.optimize_type);
1154  case VM_METHOD_TYPE_REFINED:
1155  if (def->body.orig_me) {
1156  def = def->body.orig_me->def;
1157  goto again;
1158  }
1159  else {
1160  return hash;
1161  }
1162  default:
1163  rb_bug("rb_hash_method_definition: unsupported method type (%d)\n", def->type);
1164  }
1165  return hash;
1166 }

1174 void
1175 rb_alias(VALUE klass, ID name, ID def)
1176 {
1177  VALUE target_klass = klass;
1178  rb_method_entry_t *orig_me;
1179  rb_method_flag_t flag = NOEX_UNDEF;
1180 
1181  if (NIL_P(klass)) {
1182  rb_raise(rb_eTypeError, "no class to make alias");
1183  }
1184 
1185  rb_frozen_class_p(klass);
1186  if (klass == rb_cObject) {
1187  rb_secure(4);
1188  }
1189 
1190  again:
1191  orig_me = search_method(klass, def, 0);
1192 
1193  if (UNDEFINED_METHOD_ENTRY_P(orig_me)) {
1194  if ((!RB_TYPE_P(klass, T_MODULE)) ||
1195  (orig_me = search_method(rb_cObject, def, 0),
1196  UNDEFINED_METHOD_ENTRY_P(orig_me))) {
1197  rb_print_undef(klass, def, 0);
1198  }
1199  }
1200  if (orig_me->def->type == VM_METHOD_TYPE_ZSUPER) {
1201  klass = RCLASS_SUPER(klass);
1202  def = orig_me->def->original_id;
1203  flag = orig_me->flag;
1204  goto again;
1205  }
1206 
1207  if (flag == NOEX_UNDEF) flag = orig_me->flag;
1208  rb_method_entry_set(target_klass, name, orig_me, flag);
1209 }

1212  * call-seq:
1213  * alias_method(new_name, old_name) -> self
1214  *
1215  * Makes <i>new_name</i> a new copy of the method <i>old_name</i>. This can
1216  * be used to retain access to methods that are overridden.
1217  *

1224  * end
1225  * include Mod

1229  *
1230  * Exiting with code 99
1231  */

1238  rb_print_undef_str(mod, oldname);
1239  }
1240  rb_alias(mod, rb_to_id(newname), oldid);
1241  return mod;
1242 }
