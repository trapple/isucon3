272 static ID
273 ripper_token2eventid(int tok)
274 {
275  const struct token_assoc *a;
276 
277  for (a = token_to_eventid; a->id != NULL; a++) {
278  if (a->token == tok)
279  return *a->id;
280  }
281  if (tok < 256) {
282  return ripper_id_CHAR;
283  }
284  rb_raise(rb_eRuntimeError, "[Ripper FATAL] unknown token %d", tok);
285 
286  UNREACHABLE;
287 }
